def _S_B(self, X, y):
    labels = self._labels(y)
    n_features = self._n_features(X)
    overall_mean = np.mean(X, axis=0)
    S_B = np.empty((n_features, n_features))
    for label in labels:
        _X = X[y == label]
        _mean = np.mean(_X, axis=0)
        S_B += len(_X) * \
            (_mean - overall_mean).dot((_mean - overall_mean).T)
    return S_B
